#+begin_src shell
  python3 isswisafre.py run-experiment \
          --smu-visa-address ASRL/dev/ttyUSB0::INSTR \
          --line-frequency 60 \

          --microfluidics-serial-port /dev/ttyACM0 \
          --max-gate-voltage 1.0 \
          --channel-voltage 0.05 \
          2>fd-test-01/2025-11-17/events.log  | \
